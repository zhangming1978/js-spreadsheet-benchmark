class MemoryLeakDetector {
  constructor() {
    this.baselineMemory = 0;
    this.memorySnapshots = [];
  }

  takeSnapshot(label) {
    if (!performance.memory) {
      console.warn('Performance Memory API ä¸å¯ç”¨');
      return null;
    }

    const snapshot = {
      label,
      timestamp: Date.now(),
      usedJSHeapSize: performance.memory.usedJSHeapSize,
      totalJSHeapSize: performance.memory.totalJSHeapSize,
      jsHeapSizeLimit: performance.memory.jsHeapSizeLimit
    };

    this.memorySnapshots.push(snapshot);
    return snapshot;
  }

  detectLeak() {
    if (this.memorySnapshots.length < 2) {
      return { hasLeak: false, message: 'å¿«ç…§ä¸è¶³' };
    }

    const first = this.memorySnapshots[0];
    const last = this.memorySnapshots[this.memorySnapshots.length - 1];

    const increasePercentage = (memoryIncrease / first.usedJSHeapSize) * 100;

    // å¦‚æœå†…å­˜å¢é•¿è¶…è¿‡ 20%ï¼Œå¯èƒ½å­˜åœ¨æ³„æ¼

      hasLeak: false,
      message: 'æœªæ£€æµ‹åˆ°æ˜æ˜¾çš„å†…å­˜æ³„æ¼',
      increase: memoryIncrease,
      percentage: increasePercentage
